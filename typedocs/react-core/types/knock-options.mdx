Configuration options for initializing the Knock client instance.

## Properties

<Attributes>
  <Attribute
    name="host"
    type="string"
    description="Optional custom API host for Knock. Defaults to the standard Knock API endpoint."
  />
  <Attribute
    name="logLevel"
    type="string"
    description="Optional logging level for debugging. Can be 'debug', 'info', 'warn', or 'error'."
  />
  <Attribute
    name="onUserTokenExpiring"
    type="function"
    description="Optional callback function that fires before a user token expires."
  />
  <Attribute
    name="timeBeforeExpirationInMs"
    type="number"
    description="Optional time in milliseconds before token expiration to trigger the callback. Default: 30000 (30 seconds)."
  />
</Attributes>

## Example

```typescript
const knockOptions: KnockOptions = {
  host: "https://api.knock.app",
  logLevel: "debug",
  onUserTokenExpiring: async (oldToken) => {
    // Refresh token logic
    return await refreshUserToken(oldToken);
  },
  timeBeforeExpirationInMs: 60000, // 1 minute
};
```
