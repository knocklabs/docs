Represents a connection configuration for a Microsoft Teams channel integration.

## Properties

<Attributes>
  <Attribute
    name="ms_teams_tenant_id"
    type="string"
    description="Optional Microsoft Entra tenant ID."
  />
  <Attribute
    name="ms_teams_team_id"
    type="string"
    description="Optional Microsoft Teams team ID."
  />
  <Attribute
    name="ms_teams_channel_id"
    type="string"
    description="Optional Microsoft Teams channel ID."
  />
  <Attribute
    name="ms_teams_user_id"
    type="null"
    description="User ID (typically null for channel connections)."
  />
  <Attribute
    name="incoming_webhook"
    type="object"
    description="Optional incoming webhook configuration object."
  />
</Attributes>

## Incoming Webhook Object

<Attributes>
  <Attribute
    name="url"
    type="string"
    description="The webhook URL for the Microsoft Teams channel."
  />
</Attributes>

## Example

```typescript
const connection: MsTeamsChannelConnection = {
  ms_teams_tenant_id: "12345678-1234-1234-1234-123456789012",
  ms_teams_team_id: "19:abc123def456...",
  ms_teams_channel_id: "19:def456ghi789...",
  ms_teams_user_id: null,
  incoming_webhook: {
    url: "https://outlook.office.com/webhook/...",
  },
};
```
