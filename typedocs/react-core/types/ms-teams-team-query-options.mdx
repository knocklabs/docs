Configuration options for querying Microsoft Teams teams.

## Properties

<Attributes>
  <Attribute
    name="maxCount"
    type="number"
    description="The maximum number of teams to return. Default: 1000."
  />
  <Attribute
    name="limitPerPage"
    type="number"
    description="How many teams will be returned per request. Default: 100."
  />
  <Attribute
    name="filter"
    type="string"
    description="OData $filter query parameter to filter teams. Default: null."
  />
  <Attribute
    name="select"
    type="string"
    description="OData $select query parameter to select fields. Default: 'id,displayName'."
  />
</Attributes>

## Example

```typescript
const queryOptions: MsTeamsTeamQueryOptions = {
  maxCount: 250,
  limitPerPage: 50,
  filter: "displayName eq 'Engineering'",
  select: "id,displayName,description",
};
```
