Represents a Slack channel with its basic properties.

## Properties

<Attributes>
  <Attribute
    name="name"
    type="string"
    description="The display name of the Slack channel."
  />
  <Attribute
    name="id"
    type="string"
    description="The unique identifier of the Slack channel."
  />
  <Attribute
    name="is_private"
    type="boolean"
    description="Whether the channel is private or public."
  />
  <Attribute
    name="is_im"
    type="boolean"
    description="Whether this is a direct message channel."
  />
  <Attribute
    name="context_team_id"
    type="boolean"
    description="The team ID context for the channel."
  />
</Attributes>

## Example

```typescript
const channel: SlackChannel = {
  name: "general",
  id: "C1234567890",
  is_private: false,
  is_im: false,
  context_team_id: true,
};
```
