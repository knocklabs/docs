Represents a connection configuration for a Slack channel integration.

## Properties

<Attributes>
  <Attribute
    name="access_token"
    type="string"
    description="Optional access token for the Slack integration."
  />
  <Attribute
    name="channel_id"
    type="string"
    description="Optional Slack channel ID."
  />
  <Attribute
    name="incoming_webhook"
    type="string"
    description="Optional incoming webhook URL for the Slack channel."
  />
  <Attribute
    name="user_id"
    type="null"
    description="User ID (typically null for channel connections)."
  />
</Attributes>

## Example

```typescript
const connection: SlackChannelConnection = {
  access_token: "xoxb-1234567890-abcdef...",
  channel_id: "C1234567890",
  incoming_webhook: "https://hooks.slack.com/services/...",
  user_id: null,
};
```
