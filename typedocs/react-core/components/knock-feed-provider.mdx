A React context provider that manages the notification feed state and provides feed-related functionality to child components.

## Usage

```jsx
import { KnockProvider, KnockFeedProvider } from "@knocklabs/react-core";

function App() {
  return (
    <KnockProvider apiKey="pk_test_..." userId="user-123">
      <KnockFeedProvider feedId="default">
        <NotificationFeed />
      </KnockFeedProvider>
    </KnockProvider>
  );
}
```

## Props

<Attributes>
  <Attribute
    name="feedId"
    type="string"
    description="The unique identifier for the notification feed."
  />
  <Attribute
    name="defaultFeedOptions"
    type="FeedClientOptions"
    description="Default options for configuring the feed behavior."
  />
  <Attribute
    name="children"
    type="ReactNode"
    description="Child components that will have access to the feed context."
  />
</Attributes>
