A React context provider that manages Microsoft Teams integration state and provides Teams-related functionality to child components.

## Props

<Attributes>
  <Attribute
    name="knockMsTeamsClientId"
    type="string"
    description="The Microsoft Teams client ID for your Knock Teams integration."
  />
  <Attribute
    name="redirectUrl"
    type="string"
    description="The URL to redirect to after Microsoft Teams authentication."
  />
  <Attribute
    name="children"
    type="ReactNode"
    description="Child components that will have access to the Teams context."
  />
</Attributes>

## Example

```jsx
import { KnockProvider, KnockMsTeamsProvider } from "@knocklabs/react-core";

function App() {
  return (
    <KnockProvider apiKey="pk_test_..." userId="user-123">
      <KnockMsTeamsProvider
        knockMsTeamsClientId="your-teams-client-id"
        redirectUrl="/teams/callback"
      >
        <TeamsIntegration />
      </KnockMsTeamsProvider>
    </KnockProvider>
  );
}
```
