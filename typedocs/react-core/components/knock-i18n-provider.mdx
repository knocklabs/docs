A React context provider that manages internationalization (i18n) state and provides translation functionality to child components.

## Props

<Attributes>
  <Attribute
    name="locale"
    type="string"
    description="The current locale for translations (e.g., 'en', 'fr', 'es')."
  />
  <Attribute
    name="translations"
    type="Record<string, any>"
    description="Translation dictionary containing localized strings."
  />
  <Attribute
    name="children"
    type="ReactNode"
    description="Child components that will have access to the i18n context."
  />
</Attributes>

## Example

```jsx
import { KnockProvider, KnockI18nProvider } from "@knocklabs/react-core";

const translations = {
  en: {
    "notification.new": "New notification",
    "notification.markAsRead": "Mark as read",
  },
  fr: {
    "notification.new": "Nouvelle notification",
    "notification.markAsRead": "Marquer comme lu",
  },
};

function App() {
  return (
    <KnockProvider apiKey="pk_test_..." userId="user-123">
      <KnockI18nProvider locale="en" translations={translations}>
        <NotificationUI />
      </KnockI18nProvider>
    </KnockProvider>
  );
}
```
