A React context provider that manages Slack integration state and provides Slack-related functionality to child components.

## Usage

```jsx
import { KnockProvider, KnockSlackProvider } from "@knocklabs/react-core";

function App() {
  return (
    <KnockProvider apiKey="pk_test_..." userId="user-123">
      <KnockSlackProvider
        knockSlackClientId="your-slack-client-id"
        redirectUrl="/slack/callback"
      >
        <SlackIntegration />
      </KnockSlackProvider>
    </KnockProvider>
  );
}
```

## Props

<Attributes>
  <Attribute
    name="knockSlackClientId"
    type="string"
    description="The Slack client ID for your Knock Slack integration."
    isRequired
  />
  <Attribute
    name="redirectUrl"
    type="string"
    description="The URL to redirect to after Slack authentication."
    isRequired
  />
  <Attribute
    name="children"
    type="ReactNode"
    description="Child components that will have access to the Slack context."
  />
</Attributes>
