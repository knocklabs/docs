A React context provider that initializes and provides the Knock client instance to child components throughout your application.

## Usage

```jsx
import { KnockProvider } from "@knocklabs/react-core";

function App() {
  return (
    <KnockProvider
      apiKey="pk_test_..."
      userId="user-123"
      userToken="eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9..."
    >
      <YourApp />
    </KnockProvider>
  );
}
```

## Props

<Attributes>
  <Attribute
    name="apiKey"
    type="string"
    description="Your public API key from the Knock dashboard."
  />
  <Attribute
    name="userId"
    type="string"
    description="The unique identifier for the current user."
  />
  <Attribute
    name="userToken"
    type="string"
    description="The JWT token for authenticating the user (required for production)."
  />
  <Attribute
    name="host"
    type="string"
    description="The Knock API host URL. Defaults to the production Knock API."
  />
  <Attribute
    name="children"
    type="ReactNode"
    description="Child components that will have access to the Knock client."
  />
</Attributes>
