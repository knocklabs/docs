A React context provider that initializes and provides the Knock client instance to child components throughout your application.

## Usage

```jsx
import { KnockProvider } from "@knocklabs/react-core";

function App() {
  return (
    <KnockProvider
      apiKey="pk_test_..."
      userId="user-123"
      userToken="eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9..."
    >
      <YourApp />
    </KnockProvider>
  );
}
```

## Props

<Attributes>
  <Attribute
    name="apiKey"
    type="string"
    description="Your public API key from the Knock dashboard."
    isRequired
  />
  <Attribute
    name="user"
    type="object"
    description="A user object containing information to identify the user on Knock."
  />
  <Attribute
    name="identificationStrategy"
    type="string"
    description="Defaults to `inline`. Can be set to `skip` to skip inline identification."
  />
  <Attribute
    name="userToken"
    type="string"
    description="The JWT token for authenticating the user (required when using enhanced security mode)."
  />
  <Attribute
    name="host"
    type="string"
    description="The Knock API host URL. Defaults to the production Knock API."
  />
  <Attribute
    name="logLevel"
    type="string"
    description="When set to debug, will output additional logging to aid with debugging."
  />
  <Attribute
    name="children"
    type="ReactNode"
    description="Child components that will have access to the Knock client."
  />
</Attributes>
